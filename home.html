document.addEventListener('DOMContentLoaded', function () { const carouselWrapper = document.querySelector('.carousel-wrapper'); const carouselTrack = document.querySelector('.carousel-track'); const leftButton = document.querySelector('.carousel-button.left'); const rightButton = document.querySelector('.carousel-button.right'); let scrollInterval; function startAutoScroll() { carouselWrapper.style.scrollSnapType = 'none'; // Turn off snap scrollInterval = setInterval(() => { carouselWrapper.scrollLeft += 1.5; // Small smooth move if (carouselWrapper.scrollLeft >= (carouselTrack.scrollWidth - carouselWrapper.clientWidth)) { carouselWrapper.scrollLeft = 0; } }, 10); // Very smooth } function stopAutoScroll() { clearInterval(scrollInterval); carouselWrapper.style.scrollSnapType = 'x mandatory'; // Turn snap back on } startAutoScroll(); // Pause auto-scroll on hover carouselWrapper.addEventListener('mouseenter', stopAutoScroll); carouselWrapper.addEventListener('mouseleave', startAutoScroll); // Left and right buttons leftButton.addEventListener('click', () => { carouselWrapper.scrollBy({ left: -300, behavior: 'smooth' }); }); rightButton.addEventListener('click', () => { carouselWrapper.scrollBy({ left: 300, behavior: 'smooth' }); }); // Pause scrolling when hovering buttons leftButton.addEventListener('mouseenter', stopAutoScroll); leftButton.addEventListener('mouseleave', startAutoScroll); rightButton.addEventListener('mouseenter', stopAutoScroll); rightButton.addEventListener('mouseleave', startAutoScroll); });
